CREATE TABLE Sells (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	date INTEGER NOT NULL,
	totalLiters NUMERIC NOT NULL,
	Amount NUMERIC NOT NULL,
	exchangeRate NUMERIC NOT NULL
);

CREATE TABLE SellsDetails (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	sellID INTEGER NOT NULL,
	quantity INTEGER NOT NULL,
	presentation NUMERIC NOT NULL,
	unitaryPrice NUMERIC NOT NULL,
	FOREIGN KEY (sellID) REFERENCES Sells (id)
);

CREATE TABLE Bills (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	date INTEGER NOT NULL,
	type TEXT NOT NULL,
	amount NUMERIC NOT NULL,
	supplier TEXT NOT NULL,
	exchangeRate NUMERIC NOT NULL
);

CREATE TABLE BillsDetails (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	billID INTEGER NOT NULL,
	description TEXT NOT NULL,
	quantity NUMERIC NOT NULL,
	amount NUMERIC NOT NULL,
	FOREIGN KEY (billID) REFERENCES Bills (id)
);

CREATE TABLE IncomeBs (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	sellID INTEGER NOT NULL,
	amount NUMERIC NOT NULL,
	type TEXT NOT NULL,
	FOREIGN KEY (sellID) REFERENCES Sells (id)
);

CREATE TABLE IncomeDollars (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	sellID INTEGER NOT NULL,
	amount NUMERIC NOT NULL,
	FOREIGN KEY (sellID) REFERENCES Sells (id)
);

CREATE TABLE WithdrawalBs (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	sellID INTEGER NOT NULL,
	amount NUMERIC NOT NULL,
	type TEXT NOT NULL,
	FOREIGN KEY (sellID) REFERENCES Sells (id)
);

CREATE TABLE WithdrawalDollars (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	sellID INTEGER NOT NULL,
	amount NUMERIC NOT NULL,
	FOREIGN KEY (sellID) REFERENCES Sells (id)
);
